Code Examples

JavaScript / Node.js 

// Fetch names starting with 'A', limit to 10 results
const fetchNames = async () => {
  try {
    const response = await fetch(
      'https://manning-spai.vercel.app/api/names?limit=10&alphabet=A'
    );
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.error('Error fetching names:', error);
  }
};

fetchNames();

Python 

import requests

# Fetch names starting with 'A', limit to 10 results
try:
    response = requests.get(
        'https://manning-spai.vercel.app/api/names',
        params={'limit': 10, 'alphabet': 'A'}
    )
    response.raise_for_status()  # Raises an HTTPError for bad responses
    data = response.json()
    print(data)
except requests.exceptions.RequestException as e:
    print(f"Error fetching names: {e}")
