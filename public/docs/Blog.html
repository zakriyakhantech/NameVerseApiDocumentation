API Documentation

NameVerAPI provides several endpoints to access and search through the name database. The base URL for all endpoints is https://api.nameverapi.com.

Authentication

Currently, NameVerAPI doesn't require authentication for basic usage. However, rate limiting is applied to prevent abuse.

Rate Limits

The API is currently limited to 50 requests per IP address per day. If you need higher limits for your application, please contact me to discuss your requirements.

Endpoints

Get All Names

Retrieve a list of names with optional pagination and filtering.

GET /names
GET /names?page=1&limit=20
GET /names?gender=female&origin=arabic

Search Names

Search for names by query string with optional filters.

GET /names/search?q=john
GET /names/search?q=mar&gender=unisex&limit=10

Get Name by Slug

Retrieve detailed information about a specific name using its slug.

GET /names/slug/john-doe
GET /names/slug/maria-garcia

Get Names by Origin

Retrieve names filtered by their cultural or regional origin.

GET /names/origin/english
GET /names/origin/spanish?limit=15

Code Examples

JavaScript Fetch Example

// Fetch names with search and filters
async function searchNames(query, gender, origin) {
  const params = new URLSearchParams({
    q: query,
    ...(gender && { gender }),
    ...(origin && { origin }),
    limit: 20
  });
  
  try {
    const response = await fetch(`https://api.nameverapi.com/names/search?${params}`);
    const result = await response.json();
    
    if (result.success) {
      return result.data.names;
    } else {
      console.error('API Error:', result.error);
      return [];
    }
  } catch (error) {
    console.error('Network Error:', error);
    return [];
  }
}

// Usage
searchNames('mar', 'female', 'spanish')
  .then(names => console.log(names));

React Hook Example

import { useState, useEffect } from 'react';

const useNameSearch = (query, filters = {}) => {
  const [names, setNames] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  
  useEffect(() => {
    const fetchNames = async () => {
      if (!query) {
        setNames([]);
        return;
      }
      
      setLoading(true);
      setError(null);
      
      const params = new URLSearchParams({
        q: query,
        ...filters,
        limit: 20
      });
      
      try {
        const response = await fetch(`https://api.nameverapi.com/names/search?${params}`);
        const result = await response.json();
        
        if (result.success) {
          setNames(result.data.names);
        } else {
          setError(result.error || 'Unknown API error');
        }
      } catch (err) {
        setError('Network error: ' + err.message);
      } finally {
        setLoading(false);
      }
    };
    
    fetchNames();
  }, [query, JSON.stringify(filters)]);
  
  return { names, loading, error };
};

// Usage in a component
const NameSearchComponent = () => {
  const [query, setQuery] = useState('');
  const { names, loading, error } = useNameSearch(query, { gender: 'unisex' });
  
  return (
    <div>
      <input 
        type="text"
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="Search names..."
      />
      
      {loading && <p>Loading...</p>}
      {error && <p style={{ color: 'red' }}>{error}</p>}
      
      <ul>
        {names.map(name => (
          <li key={name.id}>{name.name} - {name.origin}</li>
        ))}
      </ul>
    </div>
  );
};

Node.js Example

const axios = require('axios');

async function getNameBySlug(slug) {
  try {
    const response = await axios.get(`https://api.nameverapi.com/names/slug/${slug}`);
    
    if (response.data.success) {
      return response.data.data;
    } else {
      throw new Error(response.data.error || 'Unknown API error');
    }
  } catch (error) {
    if (error.response && error.response.status === 404) {
      console.log('Name not found');
      return null;
    }
    console.error('Error fetching name:', error.message);
    throw error;
  }
}

// Usage
getNameBySlug('john-doe')
  .then(name => {
    if (name) {
      console.log('Name details:', name);
    }
  })
  .catch(error => console.error('Error:', error));
